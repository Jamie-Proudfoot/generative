Useful commands (SLURM array job submissions)

squeue -ht pending,running -r | wc -l

array_submit *.gjf

sbatch --array=1-999 ZINC_x_y.slm

find . -size 0 -delete

cp x/*.err x_err

for i in *.err; do cp "../x/${i%.err}.out" . 2>/dev/null; done

out_to_gjf *.gjf

array_submit *.out

sbatch compress.slm
